Great choice! A Binary Search Tree (BST) is a tree where every node follows this rule:

• Left child < Parent node
• Right child > Parent node

// Python Copy ⎘
class Node:
  def __init__(self, val):
    self.val = val
    self.left = None
    self.right = None

def insert(root, val):
  if root is None:
    return Node(val)
  if val < root.val:
    root.left = insert(root.left, val)
  else:
    root.right = insert(root.right, val)
  return root
